# CP v2 M2.6 — Deep Metrics Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Enrich telemetry with per-container CPU/RAM samples from the daemon and surface them in a full-page Metrics view + aggregated sparkline on Overview deploy cards.

**Architecture:** Daemon shells out to `docker stats --no-stream --format ...` once per emission cycle, parses per-container CPU/RAM, maps container names back to component IDs, and emits them as `scope=component` samples alongside the existing host/healthcheck data. CP storage + API are unchanged (the repo already accepts arbitrary metric names). Frontend adds a `/deploys/:id/metrics` route with AreaCharts for host metrics and a small-multiples grid for component metrics, plus a tiny aggregated sparkline on each OverviewCard computed client-side from per-component `healthcheck_ok` samples.

**Tech Stack:** Go (daemon `exec`), Python (CP unchanged), React + Recharts-equivalent (`AreaChart` already ported from the prototype).

**Spec reference:** [docs/superpowers/specs/2026-04-24-control-plane-v2-vision-design.md](../specs/2026-04-24-control-plane-v2-vision-design.md), §2 ("Per-component: process CPU%, RSS...").

**Out of scope (deferred to M2.7):**
- Prometheus scrape from components declaring a `/metrics` endpoint (new ComponentSpec field + daemon collector).
- Log line-rate metric (daemon log tailer).
- Systemd unit stats (uses a different code path than Docker).

```go
package metrics

import (
	"context"
	"os/exec"
	"strconv"
	"strings"
)

// ParseDockerStats parses output from
//   docker stats --no-stream --format "{{.Name}}|{{.CPUPerc}}|{{.MemPerc}}"
// and produces samples keyed by component_id (via nameToCid mapping).
// Containers not present in nameToCid are dropped so other Docker
// containers on the host don't leak into the CP's metrics store.
func ParseDockerStats(out string, nameToCid map[string]string) []Sample {
	samples := make([]Sample, 0, len(nameToCid)*2)
	for _, line := range strings.Split(out, "\n") {
		line = strings.TrimSpace(line)
		if line == "" {
			continue
		}
		parts := strings.Split(line, "|")
		if len(parts) != 3 {
			continue
		}
		name := strings.TrimSpace(parts[0])
		if name == "" {
			continue
		}
		cid, ok := nameToCid[name]
		if !ok {
			continue
		}
		cpu, cpuOk := parsePercent(parts[1])
		ram, ramOk := parsePercent(parts[2])
		if cpuOk {
			samples = append(samples, Sample{
				Scope: "component", ScopeID: cid,
				Metric: "container_cpu_percent", Value: cpu,
			})
		}
		if ramOk {
			samples = append(samples, Sample{
				Scope: "component", ScopeID: cid,
				Metric: "container_ram_percent", Value: ram,
			})
		}
	}
	return samples
}

func parsePercent(s string) (float64, bool) {
	s = strings.TrimSpace(s)
	s = strings.TrimSuffix(s, "%")
	if s == "" {
		return 0, false
	}
	v, err := strconv.ParseFloat(s, 64)
	if err != nil {
		return 0, false
	}
	return v, true
}

// CollectDocker runs the docker CLI and returns samples for each container
// in `nameToCid`. Returns empty slice if docker is unavailable or fails.
// Caller decides what to do with the error (we log and swallow in
// PublishMetrics so a flaky docker CLI doesn't break the metrics cycle).
func CollectDocker(ctx context.Context, nameToCid map[string]string) []Sample {
	if len(nameToCid) == 0 {
		return nil
	}
	cmd := exec.CommandContext(ctx, "docker", "stats", "--no-stream",
		"--format", "{{.Name}}|{{.CPUPerc}}|{{.MemPerc}}")
	out, err := cmd.Output()
	if err != nil {
		return nil
	}
	return ParseDockerStats(string(out), nameToCid)
}
```

```go
// PublishMetrics sends a periodic event.metrics. Meant to be called from a ticker.
//
// Payload contract (v1, see CP M2 spec):
//
//	{
//	  "ts":      RFC3339 timestamp,
//	  "samples": [{scope, scope_id, metric, value}, ...],
//	}
//
// host_id is intentionally omitted — the CP fills it in from the WS
// connection's registered host_id. Host scope_id is left empty for the
// same reason (see metrics.handler).
func (o *Orchestrator) PublishMetrics(ctx context.Context, client *ws.Client) error {
	comps, err := o.Store.List(ctx)
	if err != nil {
		return err
	}

	samples := []map[string]any{}

	// Host samples (CPU%, RAM%, load1).
	for _, hs := range metrics.CollectHost(ctx) {
		samples = append(samples, map[string]any{
			"scope":    "host",
			"scope_id": "",
			"metric":   hs.Metric,
			"value":    hs.Value,
		})
	}

	// Per-component healthcheck liveness.
	nameToCid := map[string]string{}
	for _, c := range comps {
		if c.LastHCAt != nil {
			v := 0.0
			if c.LastHCOK {
				v = 1.0
			}
			samples = append(samples, map[string]any{
				"scope":    "component",
				"scope_id": c.ID,
				"metric":   "healthcheck_ok",
				"value":    v,
			})
		}
		// Build the name→id map for docker stats. We rely on the established
		// naming convention: container name is "maestro-" + component_id.
		nameToCid["maestro-"+c.ID] = c.ID
	}

	// Per-container CPU + RAM (best-effort; docker may not be available).
	for _, ds := range metrics.CollectDocker(ctx, nameToCid) {
		samples = append(samples, map[string]any{
			"scope":    ds.Scope,
			"scope_id": ds.ScopeID,
			"metric":   ds.Metric,
			"value":    ds.Value,
		})
	}

	payload := map[string]any{
		"ts":      time.Now().UTC().Format(time.RFC3339),
		"samples": samples,
	}
	return client.SendEvent(ws.TypeEventMetrics, payload)
}
```

function extractHostIds(yaml: string): string[] {
  const m = yaml.match(/\bhosts:\s*\n((?:[ \t]+\S.*\n?)+)/);
  if (!m) return [];
  const ids: string[] = [];
  for (const line of m[1].split("\n")) {
    const match = line.match(/^[ \t]+([A-Za-z0-9_.-]+)\s*:/);
    if (match) ids.push(match[1]);
  }
  return Array.from(new Set(ids));
}

function extractComponentIds(yaml: string): string[] {
  const m = yaml.match(/\bcomponents:\s*\n((?:[ \t]+\S.*\n?)+?)(?=\ndeployment:|\n\S|$)/);
  if (!m) return [];
  const ids: string[] = [];
  for (const line of m[1].split("\n")) {
    const match = line.match(/^[ \t]+([A-Za-z0-9_.-]+)\s*:/);
    if (match) ids.push(match[1]);
  }
  return Array.from(new Set(ids));

/**
 * A small component-level CPU sparkline for overview cards. Picks the first
 * component in the deploy (if any) as representative; in a future milestone
 * we'll add a true deploy-level aggregate metric computed server-side.
 */
export function DeploySparkline({ componentId }: { componentId: string | null }) {
  const enabled = !!componentId;
  const { data } = useMetricRange(
    "component", componentId || "", "container_cpu_percent",
    15 * 60, enabled,
  );
  const series = (data || []).map(([t, v]) => ({ t, v }));
  if (!enabled || series.length === 0) {
    return <div className="small dim" style={{ height: 30, display: "flex", alignItems: "center" }}>
      no samples yet
    </div>;
  }
  return <Sparkline data={series} width={280} height={30} />;
}

## Milestone Exit Criteria

1. `go test ./...` from `daemon/` is green.
2. `pytest control-plane/tests/unit/ -q` is green.
3. `npm run build` from `web-ui/` is green.
4. A running daemon on a host with Docker emits `container_cpu_percent` + `container_ram_percent` samples for any component named `maestro-<component_id>`.
5. `GET /deploys/:id/metrics` in the SPA renders per-host CPU/RAM AreaCharts and, when samples exist, per-component CPU/RAM AreaCharts.
6. Overview deploy cards render a sparkline slot (empty until first-component discovery lands in M3).

Out of scope (M2.7):
- Prometheus scrape from components declaring a `/metrics` endpoint (new `ComponentSpec.metrics.endpoint` field + daemon collector + allow-list per component).
- Log line-rate metric (daemon-side log tailer producing `log_lines_per_sec` samples).
- Systemd unit stats (different code path; can reuse `Sample` type).
- Server-side deploy-level aggregate (`healthy_fraction`) computed on the fly from component `healthcheck_ok` samples.
